滑动窗口

​ 固定长度内的最大、最小值，使用单调队列进行优化

​ 区间内最小值，

​ 序列单调上升，后来的比前面小，前面永无出头之日

​ 队列存下标

代码

#include <iostream>

using namespace std;

const int N = 1e6 + 10;

int a[N] , q[N] , n , k;

int main()
{
    cin >> n >> k;

    for(int i = 1 ; i <= n ; i++) cin >> a[i];

    int hh = 0 , tt = -1;    
    for(int i = 1 ; i <= n ; i++)
    {
        if(hh <= tt && q[hh] < i - k + 1) hh++;
        while(hh <= tt && a[i] <= a[q[tt]]) tt--;
        q[++tt] = i;

        if(i >= k) cout << a[q[hh]] << ' ';
    }

    cout << endl;
    hh = 0 , tt = -1;
    for(int i = 1 ; i <= n ; i++)
    {
        if(hh <= tt && q[hh] < i - k + 1) hh++;
        while(hh <= tt && a[i] >= a[q[tt]]) tt--;
        q[++tt] = i;

        if(i >= k) cout << a[q[hh]] << ' ';
    }
    return 0;
}

KMP

​ j= = m时，i在匹配最后的位置。

#include <iostream>
#include <string>
using namespace std;
int n,m;
const int N=1e6+100;
int ne[N];
char s[N],p[N];
int main() {
    cin>>m>>p+1>>n>>s+1;
    for(int i=2,j=0;i<=m;i++)
    {
        while(j && p[i]!=p[j+1]) j=ne[j];
        if(p[i]==p[j+1]) j++;
        ne[i]=j;
    }
    for(int i=1,j=0;i<=n;i++)
    {
        while(j && s[i]!=p[j+1]) j=ne[j];
        if(s[i]==p[j+1]) j++;
        if(j==m)
            cout<<i-j<<' ';
    }
}

最大异或对

#pragma comment(linker, "/STACK:1024000000,1024000000")
#include <stdio.h>
#include <iostream>
#include <cstdlib>
#include <cmath>
#include <cctype>
#include <string>
#include <string.h>
#include <cstring>
#include <algorithm>
#include <stack>
#include <queue>
#include <set>
#include <map>
#include <ctime>
#include <vector>
#include <fstream>
#include <list>
#include <iomanip>
#include <numeric>
using namespace std;
typedef long long ll;
typedef unsigned long long ull;
#define mst(s) memset(s, 0, sizeof(s))
const int INF = 0x3f3f3f3f;
#define LOCAL
const int N=1e7+100;
int son[N][3],cnt[N],idx;
int a[N];
int n;
void insert(int s)
{
    int p=0;
    for(int i=30;i>=0;i--)
    {
        int u=s>>i&1;
        if(!son[p][u])
            son[p][u]=++idx;
        p=son[p][u];
    }
    cnt[p]++;
}
int query(int x)
{
    int p=0;
    int res=0;
    for(int i=30;i>=0;i--)
    {
        int u=x>>i&1;
        if(son[p][!u])
        {
            res+=1<<i;
            p=son[p][!u];
        }
        else
            p=son[p][u];
    }
    return res;
}

int main() 
{
    cin>>n;
    for(int i=0;i<n;i++) cin>>a[i];
    int maxv=0;
    for(int i=0;i<n;i++)
    {
        insert(a[i]);
    }
    for(int i=0;i<n;i++)
    {
        maxv = max(maxv,query(a[i]));
    }
    cout<<maxv<<endl;
    

    
    return 0;
}

食物链

​ 带权并查积的find函数

int find(int x)
{
    if(x!=f[x])
	{
	    int p=find(f[x]);
	    d[x]+=d[f[x]];
	    f[x]=p;
	}
	return f[x];
}

#include <iostream>
#include <algorithm>
#include <string.h>
using namespace std;
const int N=1e6;
int n,f[N],d[N];
int k;
int find(int x)
{
    if(x!=f[x])
	{
	    int p=find(f[x]);
	    d[x]+=d[f[x]];
	    f[x]=p;
	}
	return f[x];
}
int main()
{
	cin>>n;
	cin>>k;
	int res=0;
	for(int i=1;i<=n;i++)
	{
		f[i]=i;
		d[i]=0;
	}
	while(k--)
	{
		int op,a,b;
		cin>>op>>a>>b;
		if(a>n || b>n) 
			res++;
		else
		{
			int fa=find(a);
			int fb=find(b);
			if(op==1)
		    {
		        if(fa!=fb)
		        {
		            f[fa]=fb;
		            d[fa]=d[b]-d[a];
		        }
		        else if((d[a]-d[b])%3) res++;
		    }
			else {
		        if(fa!=fb)
		        {
		            f[fa]=fb;
		            d[fa]=d[b]-d[a]+1;
		        }
		        else if((d[a]-d[b]-1)%3)
                    res++;
		    }
		}
		
	}
	cout<<res<<endl;
	return 0;
}

字符串哈希

#include<iostream>
using namespace std;
typedef unsigned long long ull;
const int N=1e6+100;
ull p[N],h[N];
char s[N];
ull get(int l,int r)
{
    return h[r]-(h[l-1]*p[r-l+1]);
}
int n,m;
int main()
{
    cin>>n>>m;
    cin>>s+1;
    p[0]=1;
    for(int i=1;i<=n;i++)
    {
        p[i]=p[i-1]*131;
        h[i]=h[i-1]*131+s[i]-'a';
    }
    while(m--)
    {
        int l1,r1,l2,r2;
        cin>>l1>>r1>>l2>>r2;
        if(get(l1,r1) == get(l2,r2)) cout<<"Yes"<<endl;
        else cout<<"No"<<endl;
    }
    return 0;
}
